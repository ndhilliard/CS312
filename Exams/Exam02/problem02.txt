Partition(A, low, high) {

1. pivot <- A[low]
2. i <- low + 1
3. j <- high

4. while i <= j, do

   4.1 while i <= j and A[i] <= pivot, do
      4.1.1. i <- i + 1

   4.2 while j >= i and A[j] > pivot, do
      4.2.1. j <- j - 1

   4.3 if i < j, then
      4.3.1. swap A[i] with A[j]

5. swap A[low] with A[j]

6. return j
